# Lists (variable-length arrays)

example_list = ['apples', 'oranges', 42]
example_list.append('hello world')  # appends to a list
example_list.insert(2, 'answer')  # inserts 'answer' at position 2
last_item = example_list.pop()  # removes and return last item on list
item_at_2 = example_list.pop(2)  # removes and return item at position 2
reversed_list = example_list.reverse()  # reverses a list
length_of_example_list = len(example_list)  # returns length of a list
del example_list[1]  # deletes item at index 1 (del reduces list length)
del example_list[1:2]  # deletes items from index 1 to index 2
del example_list[:]  # deletes everything from list (list now zero length)

# Dictionaries (associative arrays)

size_abbreviations = {'small': 'S', 'medium': 'M', 'large': 'L'}
my_size = size_abbreviations['medium']  # returns 'M'
size_abbreviations['extra large'] = 'XL'  # adds this element to dictionary
del size_abbreviations['medium'] # removes 'medium':'M' pair from dictionary

# Sets

a_set = set()  # creates an empty set
a_set.add('hello')  # adds to a set
a_set.remove('hello')  # removes this item from the set
a_set.clear()  # removes all elements from the set
set_from_list = set(['apples', 'oranges'])  # creates a set from a list
if 'apples' in set_from_list:  # checks to see if an item is in the set
    print('yes, we have apples!')
if 'pineapple' not in set_from_list:  # 'not' is the negation operator
    print('but no pineapples for you!')
